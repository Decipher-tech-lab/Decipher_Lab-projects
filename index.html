script.js

const display = document.getElementById('display');
const keys = document.querySelectorAll('.calculator-key');

let currentInput = '0';
let previousInput = null;
let operation = null;
let shouldResetScreen = false;

keys.forEach(key => {
    key.addEventListener('click', () => {
        const keyValue = key.textContent;
        
        if (key.classList.contains('key-clear')) {
            // Clear the calculator
            currentInput = '0';
            previousInput = null;
            operation = null;
        } else if (key.classList.contains('key-operator')) {
            // Handle operators
            if (operation !== null) calculate();
            previousInput = currentInput;
            operation = keyValue;
            shouldResetScreen = true;
        } else if (key.classList.contains('key-equals')) {
            // Calculate result
            if (operation !== null) calculate();
            operation = null;
        } else {
            // Handle numbers and decimal
            if (currentInput === '0' || shouldResetScreen) {
                currentInput = keyValue;
                shouldResetScreen = false;
            } else {
                currentInput += keyValue;
            }
        }
        
        display.value = currentInput;
    });
});

function calculate() {
    let result;
    const prev = parseFloat(previousInput);
    const current = parseFloat(currentInput);
    
    if (isNaN(prev) || isNaN(current)) return;
    
    switch (operation) {
        case '+':
            result = prev + current;
            break;
        case '-':
            result = prev - current;
            break;
        case '×':
            result = prev * current;
            break;
        case '÷':
            result = prev / current;
            break;
        default:
            return;
    }
    
    currentInput = result.toString();
    operation = null;
}

How It Works

1. HTML Structure

The calculator is built with a simple HTML structure. The main container has a display area and a grid of buttons for digits and operations.

2. CSS Styling

We use CSS Grid to create the button layout. The calculator has a clean, modern design with a dark display area and light buttons that provide visual feedback on hover.

3. JavaScript Logic

The JavaScript handles the calculator's functionality:

  • Event listeners for button clicks
  • Tracking current and previous inputs
  • Performing arithmetic operations
  • Updating the display

4. Key Features

This calculator supports basic arithmetic operations (addition, subtraction, multiplication, division), clear function, and chaining multiple operations together.
