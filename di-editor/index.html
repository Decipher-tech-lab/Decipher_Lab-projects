// Ultra-tiny DI server
import http from 'node:http';

const router = {
  GET: { '/': async (_req, res) => {
    res.writeHead(200, { 'content-type': 'application/json' });
    res.end(JSON.stringify({ ok: true, msg: 'Hello from DI' }));
  } }
};

const server = http.createServer((req, res) => {
  const route = router[req.method]?.[req.url];
  return route ? route(req, res) : (res.statusCode = 404, res.end('Not found'));
});

server.listen(8080, () => console.log('DI up on :8080'));
